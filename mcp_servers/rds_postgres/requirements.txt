# - [binary] extra 提供预编译 wheel,arm64/linux 安装无需 build-essential
# - 不用 psycopg2 / psycopg2-binary(上一代 API 设计,arm64 wheel 历史问题多)
psycopg[binary] ~= 3.2

# AWS SDK(每次 tool call 现拉 Secret)
boto3 >= 1.34

# 底层 ASGI 显式声明(让 pip resolver 给一个明确版本,不靠 mcp 间接依赖隐式锁)
starlette >= 0.40, < 1.0
uvicorn >= 0.30, < 1.0
